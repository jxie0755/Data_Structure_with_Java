SwingConstants.TRAILING 右边缘

SwingConstants.BOTTOM 靠下

四、示例

import java.awt.*;

import javax.swing.*;                    //引入SWING包

import javax.swing.ImageIcon..*;           //引入ImageIco包

public class ch4-1 {               

  public static void main(String args[]){

ImageIcon icon1=new ImageIcon("icon1.gif"); //创建两个图标

ImageIcon icon2=new ImageIcon("icon2.gif");

JLabel jl=new JLabel ("新型按钮",icon1, SwingConstants.CENTER); //创建JLabel

jl.setIconTextGap(30);                //设定图形与文字间距

JButton jb=new JButton("按这里");      //创建JButton

Jb setPressedIcon(icon2);               //设置按下后图标

JFrame f=new JFrame("JButton和Jlabel示例");  //创建JFrame框架

f.setDefaultCloseOperation(JFrame.EXIT_ON_CLOSE); //设置默认关闭窗口方法

f.getContentPane().add(jl,BorderLayout.NORTH);  //加入标签

f.getContentPane().add(jb,BorderLayout.SOUTH);  //加入按钮

f.setBounds(0,0,230,120);        //设置窗口位置和宽高

f.setVisible(true);                      //显示窗口

  }

}
